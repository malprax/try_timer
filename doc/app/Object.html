def self.time_today
  sum = 0.0
  self.today.each { |e| sum += (e.end.nil? ? Time.now : e.end) - e.start }
  sum
end

def resume!
  # only allow ended events to be resumed
  if self.end
    new_event = self.dup
    new_event.start = Time.now
    new_event.end = nil
    new_event.save
  else
    false
  end
end
